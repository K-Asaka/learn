\usepackage{amsthm}
\newtheorem{dfn}{定義} %dfnという環境を新たに定義
\newtheorem{thm}{定理} %thmという環境を新たに定義
\begin{document}
\begin{dfn}
2で割り切れる数を偶数と呼び、2で割り切れない数を奇数と呼ぶ。
\end{dfn}
\begin{thm}
偶数と奇数の積は偶数である。
\end{thm}
\begin{proof}
偶数は$2k$，奇数は$2k+1$と表せる。積を計算すると
\[ 2k(2k+1) \]
となり、$2$が因数として含まれているので、偶数である。
\end{proof}
