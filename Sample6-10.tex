\section{文章を枠で囲む}
\begin{frame}
  \frametitle{枠}
  Beamerでは枠を使用できます。
  \begin{block}{block環境}
    枠で囲まれた文章。
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{枠}
  Beamerでは他の色の枠も使用できます。
  \begin{alertblock}{alertblock環境}
    赤い枠で囲まれた文章。
  \end{alertblock}
  \begin{exampleblock}{exampleblock環境}
    緑の枠で囲まれた文章。
  \end{exampleblock}
\end{frame}

\section{箇条書きを行う}
\begin{frame}
  \frametitle{箇条書き}
  Beamerではitemizeによる箇条書きを使用できます。
  \begin{itemize}
    \item
    項目1
    \item
    項目2
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{箇条書き}
  Beamerではenumerateによる箇条書きを使用できます。
  \begin{enumerate}
    \item
    項目1
    \item
    項目2
  \end{enumerate}
\end{frame}
