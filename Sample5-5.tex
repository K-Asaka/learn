\documentclass{ltjsarticle}
\usepackage{graphicx}
\usepackage{here}
\begin{document}
widthのみを指定します。
\begin{figure}[H]
\includegraphics[width = 5cm]{mandrill.png}
\end{figure}

次に、heightのみを指定します。
\begin{figure}[H]
\includegraphics[height = 3cm]{mandrill.png}
\end{figure}

最後に、scaleを指定します。 
\begin{figure}[H]
\includegraphics[scale = 0.6]{mandrill.png}
\end{figure}
\end{document}
